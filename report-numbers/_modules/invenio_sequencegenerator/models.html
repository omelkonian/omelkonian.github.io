[docs]    def create(cls, name, template):
        """Create a new sequence."""
        placeholders = extract_placeholders(template)

        # Check that {counter} exists and double its brackets
        if not any(counter_regex.search('{{{0}}}'.format(p))
                   for p in placeholders):
            raise InvalidTemplate('No counter placeholder')
        template = double_counter(template, counter_regex)

        with db.session.begin_nested():
            # Check if parent sequence exists
            try:
                parent = Sequence.query.filter(
                    Sequence.name.in_(placeholders)
                ).one_or_none()
            except MultipleResultsFound:
                raise InvalidTemplate('More than 1 parents in template')

            seq = cls(
                name=name,
                template=template,
                parent=parent.name if parent else None
            )
            db.session.add(seq)

            return seq

        """Get the sequence with the given name."""

[docs]    def next_or_create(cls, template):
        """Get the next report number for the given template."""
        with db.session.begin_nested():
            rec = db.session.query(cls).filter_by(template=template).first()

            if rec is None:
                rec = TemplatedCounter.create(template)
            else:
                rec.counter += 1

            return rec.template.format(counter=rec.counter).\
                encode('ascii', 'ignore')



[docs]    def reset(self, counter=0):
        """Reset template's counter."""
        with db.session.begin_nested():

            # Ensure no children exist
            formatted = self.template.format(counter=self.counter)
            if db.session.query(TemplatedCounter).filter(
                    TemplatedCounter.template.contains(formatted)
            ).count() > 0:
                raise InvalidCall('Cannot reset sequence: Children exist')

            self.counter = counter
