[docs]class SequenceIterator:
    """Iterator for sequences."""

    def __init__(self, sequence_name, **kwargs):
        """Initialize iterator with the enclosing sequence and fixed kwargs.

        :param sequence_name The name of the sequence to iterate.
        :param kwargs The formatting arguments of the sequence's template.
        """
        self.sequence = Sequence.get(sequence_name)
        self.kwargs = kwargs

    def __iter__(self):
        """Enable iterative capabilities."""
        return self
