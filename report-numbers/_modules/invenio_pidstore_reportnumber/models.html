counter_regex = re.compile(r'({counter(!.)?(:.*)?})')

    """Represents sequence definitions."""

    __tablename__ = 'sequencegenerator_sequence'

    name = db.Column(db.String, primary_key=True)

    template = db.Column(db.String, unique=True)
    """The template of the sequence."""

    parent = db.Column(ForeignKey(name))
    """Indicates if the template depends on another one."""

    @classmethod

[docs]    def create(cls, name, template):
        """Create a new sequence."""
        placeholders = extract_placeholders(template)

        # Check that {counter} exists and double its brackets
        if all(map(
                lambda p: not counter_regex.search(
                    '{}{}{}'.format('{', p, '}')
                ), placeholders
        )):
            raise InvalidTemplate('No counter placeholder')
        template = double_counter(template)

        # Check if parent sequence exists
        names = [seq.name for seq in Sequence.query.all()]
        parents = filter(lambda p: p in names, placeholders)
        if len(parents) > 1:
            raise InvalidTemplate('More than 1 parents in template')

        parent = parents[0] if parents else None

        db.session.add(cls(
            name=name,
            template=template,
            parent=parent
        ))

[docs]    def next(cls, id, **kwargs):
        """Get the next report number for the given sequence."""
        with db.session.begin_nested():
            seq = db.session.query(cls).filter_by(name=id).first() or \
                  db.session.query(cls).filter_by(template=id).first()
            if seq is None:
                raise NoSuchSequence()

            try:
                template = seq.template.format(**kwargs)
            except KeyError:
                raise InvalidCall('Improper arguments for string formatting')

            if seq.parent:
                # Check if parent's report number exists in PID store
                try:
                    get_distribution('invenio-pidstore')
                    # from invenio_pidstore.errors import PIDDoesNotExistError
                    from invenio_pidstore.models import PersistentIdentifier

                    try:
                        PersistentIdentifier.query.filter_by(
                            pid_value=template
                        ).one()
                    except NoResultFound:
                        raise InvalidCall('Parent PID does not exist')
                except DistributionNotFound:
                    pass

                return ReportNumber.next_or_create(template)
            else:
                return ReportNumber.next_or_create(template)

                         primary_key=True, index=True)

    counter = db.Column(db.Integer, nullable=False)

    @classmethod

[docs]    def create(cls, template, counter=1):
        """Create a new report number."""
        with db.session.begin_nested():
            rec = cls(
                template=template,
                counter=counter
            )
            db.session.add(rec)
        return rec


    @classmethod

[docs]    def next_or_create(cls, template):
        """Get the next report number for the given template."""
        with db.session.begin_nested():
            rec = db.session.query(cls).filter_by(template=template).first()

            if rec is None:
                rec = ReportNumber.create(template)
            else:
                rec.counter += 1

        return rec.template.format(counter=rec.counter)

[docs]def get_next(name_or_template, **kwargs):
    """Return the next available report number for the given template."""
    next = Sequence.next(name_or_template, **kwargs).encode('ascii', 'ignore')

    return next

[docs]def extract_placeholders(template):
    """Extract the template's placeholder names."""
    return re.findall(r'{(.*?)}', template)




[docs]def double_counter(template):
    """Double brackets around the 'counter' for 2-step formatting."""
    return re.sub(counter_regex, r'{\1}', template)
